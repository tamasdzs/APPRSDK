157  void OrthonormalHermite<T>::setOrtPolynomials()
158  {
159  const unsigned int m = this->_domain.cols();
160  const unsigned int n = this->_degrees;
161  double pi = 4.0*atan(1.0);
162  MatHelper<T> helper;
163 
164  EAColVec<T> x = this->_domain.transpose().array();
165  EArray<T> H = this->setClassicalHermitePolynomials(x, n);
166  EArray<T> DH = EArray<T>::Zero(m, n);
167 
168  EAColVec<T> w = ((T)(-1)*(x*x)/(T)(2.0)).exp();
169  EAColVec<T> dw = (T)(-1)*x*w;
170 
171  H.block(0, 0, m, 1) = w*H.block(0, 0, m, 1)/sqrt(pow(2, (1-1))*helper.Factorial(1-1)*sqrt(pi));
172  DH.block(0, 0, m, 1) = dw/sqrt(pow(2, (1-1))*helper.Factorial(1-1)*sqrt(pi));
173 
174  for (unsigned int i = 2; i <= n; ++i)
175  {
176  H.block(0, i-1, m, 1) = w*H.block(0, i-1, m, 1)/sqrt(pow(2, (i-1))*helper.Factorial(i-1)*sqrt(pi));
177  DH.block(0, i-1, m, 1) = sqrt(2*(i-1))*H.block(0, i-2, m, 1) -1*x*H.block(0, i-1, m, 1);
178  }

181  this->_dFunctionSystem = DH.matrix();
