374 void VariableProjection<T>::formJacobian()
375 {
376  //TODO:
377  //1. typedef a hosszu kiirasok helyett.
378  //2. std tarolok helyett hasznalj eigen-t -> ne kelljenek a convert fuggvenyek
379  //3. Ha megis kell konvertalni, akkor ne adj at membereket pl. convertVecToMat(_nonLinParams) ne legyen

381  EMatrix<T> funSys = _functionSystem->GetFunctionSystem();
382  EMatrix<T> dPhi = _functionSystem->GetPartialDerivativesFunctionSystem();
383  EMatrix<T> s;
384 
385  int rank;
386 
387  // Get SVN decomposition
388  Eigen::BDCSVD<EMatrix<T>> svd(_weights*funSys, Eigen::ComputeFullU | Eigen::ComputeFullV);
389 
390  if (_weights.cols() > 1) // n > 1
391  {
392  s = svd.singularValues();
393  }
394  else if (_weights.cols() == 1) // TODO: make sure dimensions are okay
395  {
396  s = svd.singularValues().asDiagonal();
397  }
398  else // n == 0
399  {
400  if (_functionSystem->GetIndex().size() == 0)
401  {
402  _jacobian.resize(0, 0);
403  }
404  else
405  {
406  _jacobian.resize(_signal.cols(), _nonLinParams.cols());
407  _jacobian = (EArray<T>::Zero(_signal.cols(), _nonLinParams.cols())).matrix();
408 
409  for (unsigned int i = 0; i < _jacobian.rows(); ++i)
410  {
411  for (unsigned int j = 0; j < _functionSystem->GetIndex().cols(); ++j)
412  {
413  _jacobian(i, _functionSystem->GetIndex()(1, j)) = (-1*_weights*dPhi)(0,0);
414  }
415  }
416  }
417 
418  _linParams.resize(1,0);
419  _approximation = funSys;
420  _weighedResidual = ((_weights*(_signal - _approximation).transpose()).array().abs()).matrix();
421  return;
422  }
423 
424  rank = svd.singularValues().rows();
425 
426  EMatrix<T> temp = svd.matrixU().block(0,0, svd.matrixU().rows(), rank).transpose()*(_weights*_signal.transpose());
427 
428  _linParams = ((svd.matrixV().block(0, 0, svd.matrixV().rows(), rank)) * ((temp.array() / s.array()).matrix())).transpose();
429  _approximation = funSys * _linParams.transpose();
430 
431  _weighedResidual = ((_weights*(_signal - _approximation).transpose()).array().abs()).matrix();
432  _currentError = _weighedResidual.norm();
433 
434  // Form the Jacobian
435  EMatrix<T> wdPhi = _weights*_functionSystem->GetPartialDerivativesFunctionSystem();
436  ERowVec<T> wdPhiResid = (wdPhi.transpose() * _weighedResidual.transpose()).transpose();
437  EMatrix<T> T2;
438  EMatrix<T> Jac1;
439 
440  Jac1.resize(_approximation.cols(), _nonLinParams.cols());
441  T2.resize(_linParams.cols(), _nonLinParams.cols());
442 
443  for (unsigned int i = 0; i < _nonLinParams.cols(); ++i)
444  {
445  std::vector<int> range;
446 
447  for (unsigned int j = 0; j < _functionSystem->GetIndex().cols(); ++j)
448  {
449  if (_functionSystem->GetIndex()(1, j) == i)
450  {
451  range.push_back(j);
452  }
453  }
454 
455  ERowVec<T> indrows;
456  indrows.resize(range.size());
457 
458  for (unsigned int j = 0; j < range.size(); ++j)
459  {
460  indrows(j) = _functionSystem->GetIndex()(0, range[j]);
461  }
462

464  jac1WdPhi.resize(wdPhi.rows(), range.size());
465 
466  for (unsigned int j = 0; j < jac1WdPhi.rows(); ++j)
467  {
468  for (unsigned int k = 0; k < jac1WdPhi.cols(); ++k)
469  {
470  jac1WdPhi(j , k) = wdPhi(j, range[k]);
471  }
472  }
473 
474  ERowVec<T> cTemp;
475  cTemp.resize(indrows.cols());
476 
477  for (unsigned int j = 0; j < cTemp.cols(); ++j)
478  {
479  cTemp(j) = _linParams(indrows(j));
480  }
481 
482  ERowVec<T> jac1IthColumn = (jac1WdPhi*cTemp.transpose()).transpose();
483 
484  for (unsigned int j = 0; j < Jac1.rows(); ++j)
485  {
486  Jac1(j, i) = jac1IthColumn(j);
487  }
488 
489  ERowVec<T> wdPhiRrange;
490  wdPhiRrange.resize(range.size());
491 
492  for (unsigned int j = 0; j < range.size(); ++j)
493  {
494  wdPhiRrange(j) = wdPhiResid(range[j]);
495  }
496 
497  for (unsigned int j = 0; j < indrows.cols(); ++j)
498  {
499  T2(indrows(j), i) = wdPhiRrange(j);
500  }
501  }
502 
503  Jac1 = Jac1 - svd.matrixU().block(0, 0, svd.matrixU().rows(), rank) * (svd.matrixU().block(0, 0, svd.matrixU().rows(), rank).transpose() * Jac1);
504 
505  EMatrix<T> beg;
506  beg.resize(s.rows(), s.rows());
507 
508  for (unsigned int i = 0; i < s.rows(); ++i)
509  {
510  for (unsigned int j = 0; j < s.rows(); ++j)
511  {
512  if (i == j)
513  {
514  beg(i,j) = 1/s(i, 0);
515  }
516  else
517  {
518  beg(i,j) = 0;
519  }
520  }
521  }
522 
523  EMatrix<T> middle = svd.matrixV().block(0,0, svd.matrixV().rows(), rank).transpose();
524  EMatrix<T> end = T2.block(0,0, _linParams.cols(), T2.cols());
525 
526  T2 = beg*(middle.transpose()*end);
527 
528  Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>Jac2 = svd.matrixU().block(0,0, svd.matrixU().rows(), rank) * T2;
529 
530  _jacobian = -1*(Jac1 + Jac2);
531 }
